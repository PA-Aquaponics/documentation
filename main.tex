\makeatletter
\def\@maketitle{%
	\newpage
	\null
	\vskip 1cm%
	\begin{center}%
		\let \footnote \thanks
		{\LARGE \@title \par}%
		\vskip 2cm%
		{\large
			\lineskip .25em%
			\begin{tabular}[t]{c}%
				\@author
			\end{tabular}\par}%
		\vfill%
		{\large \@date}
	\end{center}%
	\par
	\vskip 1.5em
}
\makeatother
